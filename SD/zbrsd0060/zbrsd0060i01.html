MODULE USER_COMMAND_0100 INPUT.
  CASE OK_CODE.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'CREATE'.
      PERFORM CHECK_CREATE.
    WHEN 'DELETE'.
      PERFORM CHECK_ROW USING '삭제'.

    WHEN 'UPDATE'.
      PERFORM CHECK_ROW USING '수정'.
  ENDCASE.
ENDMODULE.
